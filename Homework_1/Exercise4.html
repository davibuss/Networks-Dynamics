# Initialization of the number of iterations, 50 is the chosen number
n_iter = 50;


#I need to save the final opinion in a dictionary, whose key is represented by the position (i,j,k) of
#the 1-stubborn agent
#Same for average opinion, always a dictionary initialized in the same way of final opinion
final_opinions = dict()
average_opinion = dict() 


for (i,j,k) in cube.nodes:
    # In my case the position (0,0,0) is taken by the 0-stubborn node
    if (i,j,k)==(0,0,0): #If the generic node is equal to (0,0,0) I have to do anything
        #because it is the starting point
        continue
        
    # Stubborn and regular nodes
    #Define the list of stubborn node, surely there is the fixed node (0,0,0) and a generic node which acts as 
    #stubborn (i,j,k)
    stubborn = [(0,0,0), (i,j,k)];
    #I need to keep trace of their identifier
    stubborn_id = [indices.get(key) for key in stubborn]
    #A regular node is a node not stubborn, I have also to list them
    regular = [node for node in cube.nodes if node not in stubborn]
    regular_id = [id for id in range(n_nodes) if id not in stubborn_id]
    print("Stubborn nodes:", stubborn)

    # Input vector to stubborn node, "0" is associated to (0,0,0), "1" to each generic node (i,j,k)
    u = [0,1]


    # Thee aim is to compute the P matrix by exploting linear algebra tools, thanks to Numpy
    A = nx.adjacency_matrix(cube) 
    A = A.toarray() 
    degrees = np.sum(A,axis=1)
    D = np.diag(degrees)
    P = np.linalg.inv(D) @ A

    # I need the submatrices Q and E of P
    Q = P[np.ix_(regular_id, regular_id)]
    E = P[np.ix_(regular_id, stubborn_id)]

    # I define an initial condition to regular nodes, I have decided to put them randomly also to demonstrate
    #that the initial condition does not matter a lot in the simulation. On top of that, no indication on 
    #general condition were given and so I preferred to be as more general as possible.
    ic = np.random.uniform(0,1,len(regular))
    #Now I have all the elements to build dynamics model
    # Initial condition phase
    x = np.zeros((n_nodes,n_iter))
    x[stubborn_id,0] = u;
    x[regular_id,0] = ic;
    #Average dynamics algorithm computation during 50 iterations
    for t in range(1,n_iter):
        x[regular_id, t] = Q @ x[regular_id, t-1] + E @ x[stubborn_id, t-1]
        x[stubborn_id, t] = x[stubborn_id, t-1];
    # I store the final opinions and average opinion on the already created dictionaries, with position as key
    final_opinions[(i,j,k)] = x[:,n_iter-1]
    average_opinion[(i,j,k)] = np.average(final_opinions[(i,j,k)])
    print("Average opinion:", average_opinion[(i,j,k)])
